lw 0 1 num
lw 0 2 denom
lw 0 3 count
lw 0 4 sum
lw 0 5 one

beq 2 0 done # กรณี ถ้าตัวหารเป็น 0
loop
        add 2 4 4
        add 5 3 3   #  จะเก็บผลลัพธ์ไว้ที่ reg 3
        beq 4 1 done
        beq 0 0 loop

done
        halt    # สิ้นสุดโปรแกรม

count . fill 0  # ตัวนับ
sum .fill 0     # ผลลัพธ์รวม
num .fill 100      # เลขตั้งต้น
denom .fill 50   # เลขตัวหาร
one .fill 1
